            count <= 4'b0000;

! Fortran: Numerical Computing
SUBROUTINE matrix_multiply(A, B, C, n)
    IMPLICIT NONE
    INTEGER, INTENT(IN) :: n
    REAL, INTENT(IN) :: A(n,n), B(n,n)
    REAL, INTENT(OUT) :: C(n,n)
    INTEGER :: i, j, k
    
    DO i = 1, n
        DO j = 1, n
            C(i,j) = 0.0
            DO k = 1, n
                C(i,j) = C(i,j) + A(i,k) * B(k,j)
            END DO
        END DO
    END DO
END SUBROUTINE
